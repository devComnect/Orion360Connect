SELECT
    dal.id AS acesso_id,
    u.id AS usuario_id,
    u.name AS nome_usuario,
    d.id AS dispositivo_id,
    d.name AS nome_dispositivo,
    FROM_UNIXTIME(dal.time) AS data_acesso,
    dal.event AS tipo_evento,
    dal.hw_device_id,
    dal.identifier_id,
    dal.card_value,
    dal.qrcode_value
FROM
    door_access_logs dal
LEFT JOIN
    users u ON dal.user_id = u.id
LEFT JOIN
    devices d ON dal.device_id = d.id
ORDER BY
    dal.time DESC;
